import Rpi.GPIO as GPIO

GPIO.setmode(GPIO.BOARD)
GPIO.steup(8,GPIO.IN)
GPIO.setup(36,GPIO.OUT)

while True:
    val=GPIO.input(8)
    print(val)
    if val==1:
        GPIO.output(36,GPIO.LOW)
        print("Not Detecting.....")
    else:
        GPIO.output(36,GPIO.HIGH)
        print("Detected !!")
